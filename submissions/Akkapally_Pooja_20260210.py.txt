# float object demo program 

1.	a = 10.8 
print(a) ------- 10.8
print(type(a)) -----<class ‘float’>
print(id(a)) -----address of object 10.8
b = 25.
print(b) -----25.0
print(type(b)) -----<class ‘float’>
c = .689
print(c) ------ 0.689
d = 3.4E2
print(d) ----340.0
print(type(d)) -----<class ‘float’>
e = 9.62e-2
print(e)------0.0962
print (9.8.2) --------error (2 decimal points)
