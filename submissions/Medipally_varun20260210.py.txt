# complex object demo program
a = 3 + 4j
print(a) # 3+4j 
print(type(a)) # class 'complex'
print(id(a)) # address of a 
print(a . real) # 3.0
print(a . imag) # 4.0
print(type(a . real)) # class 'float'
print(type(a . imag)) # class 'float'
